CREATE TABLE cliente (
    -- Dados Pessoais
    id_cliente INT PRIMARY KEY AUTO_INCREMENT, -- Chave primária e auto-incremento para identificação única
    nome VARCHAR(100) NOT NULL,
    data_nascimento DATE,
    cpf VARCHAR(14) UNIQUE NOT NULL, -- Geralmente no formato '000.000.000-00'
    rg VARCHAR(20) UNIQUE,
    estado_civil VARCHAR(20),

    -- Endereço
    cep VARCHAR(10), -- Geralmente no formato '00000-000'
    rua VARCHAR(100),
    numero VARCHAR(10),
    bairro VARCHAR(100),
    cidade VARCHAR(100),
    estado CHAR(2), -- Usando 2 caracteres para a sigla do estado

    -- Contato
    telefone VARCHAR(15), -- Telefone fixo
    celular VARCHAR(15),
    email VARCHAR(100) UNIQUE
);
